CREATE TABLE USERS(
USERNAME VARCHAR(50),
EMAIL VARCHAR(50),
PASSWORD VARCHAR(50),
VERIFICATION_STATUS BOOLEAN DEFAULT FALSE,
USER_ID VARCHAR(10) PRIMARY KEY
);

CREATE TABLE COLLEGES(
NAME VARCHAR(50),
LOCATION VARCHAR(50),
WEBSITE VARCHAR(50)
);

CREATE TABLE VER_REQ(
VERFICATION_ID VARCHAR(50),
USER_ID VARCHAR(50),
FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)	
);
